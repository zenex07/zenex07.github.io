squareft_rag.py 
# SquareFt AI — RAG Pipeline
from langchain import LLM, RAG
from chromadb import Client
from unsloth import FastLanguageModel

# Load fine-tuned Qwen 8B with LoRA
model, tokenizer = FastLanguageModel.from_pretrained(
    model_name = "qwen-8b-squareft-lora",
    max_seq_length = 4096,
    load_in_4bit = True,
)

# Initialize ChromaDB for property data
vector_db = Client().get_collection(
    name="indore_properties"
)

# Hinglish query processing
def ask_squareft(query: str):
    context = vector_db.query(query, n=5)
    return model.generate(
        prompt=f"Context: {context}\nQuery: {query}"
    )
